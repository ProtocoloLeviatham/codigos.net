Análisis de un script de Python simple

Este es un ejemplo de cómo mostrar código. El siguiente script escanea puertos abiertos en una IP local. Útil para auditorías de red básicas.


import socket

def scan_port(port):
    try:
        s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        s.settimeout(0.1)
        if s.connect_ex(('127.0.0.1', port)) == 0:
            print(f"[+] Puerto {port} abierto")
        s.close()
    except Exception as e:
        pass

for port in range(1, 1025):
    scan_port(port)
